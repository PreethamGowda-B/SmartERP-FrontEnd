import type { MetadataRoute } from "next"

export default function robots(): MetadataRoute.Robots {
  return {
    rules: {
      userAgent: "*",
      allow: "/",
      disallow: ["/owner", "/employee", "/backend-test"],
    },
    sitemap: "https://smart-erp-front-end.vercel.app/sitemap.xml",
  }
}
